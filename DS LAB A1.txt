// implement basic array operations:insertion.deltion.searchingand traversal
#include <stdio.h>

int main()
{
int arr[100],n,i,pos,value,choice,found;
printf("enter number of elements:");
scanf("%d",&n);
printf("enter elements:\n");
for(i=0;i<n;i++) {
    scanf("%d",&arr[i]);
}
do {
    printf("\n---MENU---\n");
    printf("1.Traversal\n2.insertion\n3.deletion\n4.searching\n5.exit\n");
    printf("enter your choice:");
    scanf("%d",&choice);
    switch(choice){
 case1:
        printf("array elements are:");
    for(i=0;i<n;i++) {
        printf("%d",arr[i]);
    }
    break;
 case2: 
    printf("enter position and value:");
    scanf("%d%d",&pos,&value);
    for(i=n;i>=pos;i--){
        arr[i]=arr[i-1];
    }
    arr[pos-1]=value;
    }
    n++;
    break;
case3:
    printf("enter position to delete:");
    scanf("%d",&pos);
    for(i=pos-1;i<n;i++){
        arr[i]=arr[i+1];
    }
    n--;
break;
case4:
printf("enter elemet to search:");
scanf("%d",&value);
found=0;
for(i=0;i<n;i++) {
    if(arr[i]==value) {
        printf("element found at position%d\n",i++);
        found=1;
        break;
        }
}
if(found==0)
printf("element not found\n");
break;
case5:
printf("exiting...\n");
break;

default:
printf("invalid choice\n");
  }
}
while(choice!=5);
return 0;
}
